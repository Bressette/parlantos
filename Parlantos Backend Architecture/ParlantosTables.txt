create table server(
    id bigint,
    created_at timestamp,
    name varchar,
    description varchar,
    image varchar,
    primary key(id)
);
create table text_channel(
    id bigint,
    created_at timestamp,
    server_id bigint,
    title varchar,
    description varchar,
    primary key(id),
    foreign key(server_id) references server(id)
);
create table member(
    id bigint,
    username varchar,
    password varchar,
    salt varchar,
    display_name varchar,
    name_tag varchar,
    primary key(id)
);
create table message(
    id bigint,
    created_at timestamp,
    content varchar,
    member_id bigint,
    text_channel_id bigint,
    foreign key(member_id) references member(id),
    foreign key(text_channel_id) references text_channel(id),
    primary key(id)
);
create table reply(
    id bigint,
    created_at timestamp,
    content varchar,
    member_id bigint,
    message_id bigint,
    foreign key(message_id) references message(id),
    foreign key(member_id) references member(id),
    primary key(id)
);
create table reaction(
    id bigint,
    created_at timestamp,
    content varchar,
    message_id bigint,
    foreign key(message_id) references message(id),
    primary key(id)
);
create table pins(
    id bigint,
    created_at timestamp,
    message_id bigint,
    server_id bigint,
    text_channel_id bigint,
    foreign key(message_id) references message(id),
    foreign key(server_id) references server(id),
    foreign key(text_channel_id) references text_channel(id),
    primary key(id)
);
create table voice_channel(
    id bigint,
    created_at timestamp,
    server_id bigint,
    title varchar,
    foreign key(server_id) references server(id),
    primary key(id)
);
create table friend(
    id bigint,
    member_one_id bigint,
    member_two_id bigint,
    foreign key(member_one_id) references member(id),
    foreign key(member_two_id) references member(id),
    primary key(id)
);
create table server_member(
    id bigint,
    member_id bigint,
    server_id bigint,
    foreign key(member_id) references member(id),
    foreign key(server_id) references server(id),
    primary key(id)
);
create table role(
    id bigint,
    name varchar,
    primary key(id)
);
create table server_role(
    id bigint,
    server_member_id bigint,
    role_id bigint,
    foreign key(server_member_id) references server_member(id),
    foreign key(role_id) references role(id),
    primary key(id)
);
create table permission(
    id bigint,
    name varchar,
    description varchar,
    primary key(id)
);
create table role_permission(
    id bigint,
    role_id bigint,
    permission_id bigint,
    primary key(id),
    foreign key(role_id) references role(id),
    foreign key(permission_id) references permission(id)
);
create table discord_text_channel(
	discord_text_channel_id bigint,
	text_channel_id bigint,
	created_at timestamp,
	getMessages boolean,
	postMessages boolean,
	foreign key(text_channel_id) references text_channel(id),
	primary key(discord_text_channel_id, text_channel_id)
);
create table discord_message(
	message_id bigint,
	discord_message_id bigint,
	created_at timestamp,
	foreign key(message_id) references message(id),
	primary key(message_id, discord_message_id)
);
